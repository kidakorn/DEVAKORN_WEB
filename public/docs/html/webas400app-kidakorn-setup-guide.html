3
Create js/config.js — your ".env" for the browser
สร้าง config.js — ทำหน้าที่แทน .env

A real .env file can't be read by browser JavaScript. This file plays that role instead — it's the single place you edit when switching environments.

const ENVIRONMENTS = {
  local: {
    BASE_URL: "/api/web/serveices"
  },
  production: {
    BASE_URL: "/api/web/serveices"   // 🔴 edit this when the real path is different in production
  }
};

function detectEnvironment() {
  const host = window.location.hostname;
  return (host === "localhost" || host === "127.0.0.1") ? "local" : "production";
}

const IWS_CONFIG = {
  ENV: detectEnvironment(),
  BASE_URL: ENVIRONMENTS[detectEnvironment()].BASE_URL,

  // เพิ่ม endpoint ใหม่ตรงนี้ทุกครั้งที่แอดมิน deploy service เพิ่ม
  ENDPOINTS: {
    GET_DATA:    "/yourservice/GET_DATA",
    ADD_DATA:    "/yourservice/ADD_DATA",
    UPDATE_DATA: "/yourservice/UPDATE_DATA",
    DELETE_DATA: "/yourservice/DELETE_DATA"
  },

  AUTH: {
    enabled: false,     // 🔴 set true if the API requires login
    username: "",
    password: ""
  }
};

console.log("[IWS_CONFIG]", IWS_CONFIG.ENV, IWS_CONFIG.BASE_URL);
🔴 PRODUCTION CHANGE: Update ENVIRONMENTS.production.BASE_URL if the live server uses a different path/port than dev. If the API needs login, set AUTH.enabled = true and fill in credentials (see Step 8 for a safer alternative).
ภาษาไทย: ไฟล์นี้คือจุดเดียวที่ต้องแก้ตอนย้ายจาก dev ไป production — ปกติแก้แค่ BASE_URL กับ AUTH เท่านั้น ส่วนที่เหลือไม่ต้องแตะ (แก้ชื่อ endpoint ใน ENDPOINTS ให้ตรงกับที่แอดมิน deploy จริง)
4
Create js/api.js — all your fetch calls in one place
สร้าง api.js — รวมฟังก์ชันเรียก API ไว้ที่เดียว
function buildHeaders() {
  const headers = { "Content-Type": "application/json" };
  if (IWS_CONFIG.AUTH.enabled) {
    headers["Authorization"] = "Basic " + btoa(IWS_CONFIG.AUTH.username + ":" + IWS_CONFIG.AUTH.password);
  }
  return headers;
}

async function handleResponse(res) {
  const raw = await res.text();
  if (!res.ok) throw new Error(`HTTP ${res.status}: ${raw}`);
  try { return JSON.parse(raw); }
  catch(e) { throw new Error("Response is not valid JSON: " + raw); }
}

const IWS_API = {
  async getData() {
    const res = await fetch(IWS_CONFIG.BASE_URL + IWS_CONFIG.ENDPOINTS.GET_DATA, {
      method: "GET", headers: buildHeaders()
    });
    return handleResponse(res);
  },
  async addData(payload) {
    const res = await fetch(IWS_CONFIG.BASE_URL + IWS_CONFIG.ENDPOINTS.ADD_DATA, {
      method: "POST", headers: buildHeaders(), body: JSON.stringify(payload)
    });
    return handleResponse(res);
  },
  async updateData(payload) {
    const res = await fetch(IWS_CONFIG.BASE_URL + IWS_CONFIG.ENDPOINTS.UPDATE_DATA, {
      method: "PUT", headers: buildHeaders(), body: JSON.stringify(payload)
    });
    return handleResponse(res);
  },
  async deleteData(id) {
    const res = await fetch(IWS_CONFIG.BASE_URL + IWS_CONFIG.ENDPOINTS.DELETE_DATA, {
      method: "DELETE", headers: buildHeaders(), body: JSON.stringify({ id })
    });
    return handleResponse(res);
  }
};
ภาษาไทย: ไฟล์นี้ปกติไม่ต้องแก้ตอนขึ้น production — แก้แค่ตอนมี endpoint ใหม่เพิ่มเข้ามาเท่านั้น ต้องโหลดไฟล์นี้ต่อจาก config.js เสมอ
5
Create index.html and load both scripts
สร้าง index.html แล้วโหลดสคริปต์ทั้งสองไฟล์
<!DOCTYPE html>
<html lang="th">
<head>
<meta charset="UTF-8">
<meta name="color-scheme" content="light">
<title>webas400app/kidakorn</title>
<link rel="stylesheet" href="css/style.css?v=1">
</head>
<body>
  <div class="wrap">
    <h1>My Page</h1>
    <button id="testBtn">Test IWS</button>
    <div id="resultArea"></div>
  </div>

  <!-- ต้องโหลด config.js ก่อน api.js เสมอ -->
  <script src="js/config.js?v=1"></script>
  <script src="js/api.js?v=1"></script>
  <script>
    document.getElementById('testBtn').addEventListener('click', async () => {
      const area = document.getElementById('resultArea');
      try {
        const data = await IWS_API.getData();
        area.textContent = JSON.stringify(data, null, 2);
      } catch (err) {
        area.textContent = 'Error: ' + err.message;
      }
    });
  </script>
</body>
</html>
🔴 PRODUCTION CHANGE: Bump the ?v=1 number every time you edit config.js, api.js, or style.css — otherwise browsers may keep serving a cached (old) version.
ภาษาไทย: ต้องโหลด config.js ก่อน api.js เสมอ (ลำดับสำคัญ) และอย่าลืมไล่เลข ?v= ทุกครั้งที่แก้ไฟล์ ไม่งั้นเบราว์เซอร์จะ cache ของเก่าไว้

6
Check web.config — add 2 things if missing
เช็ก web.config — เพิ่ม 2 อย่างถ้ายังไม่มี
⚠️ This is a reference to merge into your real web.config — never overwrite it directly, and never delete the admin's existing rewrite rule.

A. Don't rewrite real static files

<rule name="YourExistingRule" stopProcessing="true">
  <match url="^(.*)$" />
  <conditions logicalGrouping="MatchAll">
    <add input="{REQUEST_FILENAME}" matchType="IsFile" negate="true" />
    <add input="{REQUEST_FILENAME}" matchType="IsDirectory" negate="true" />
  </conditions>
  <action type="Rewrite" url="..." />
</rule>

B. Allow .json files (skip if mimeMap already exists)

<system.webServer>
  <staticContent>
    <mimeMap fileExtension=".json" mimeType="application/json" />
  </staticContent>
</system.webServer>

Full reference (both parts combined)

<configuration>
  <system.webServer>

    <rewrite>
      <rules>
        <rule name="YourExistingRule" stopProcessing="true">
          <match url="^(.*)$" />
          <conditions logicalGrouping="MatchAll">
            <add input="{REQUEST_FILENAME}" matchType="IsFile" negate="true" />
            <add input="{REQUEST_FILENAME}" matchType="IsDirectory" negate="true" />
          </conditions>
          <action type="Rewrite" url="{admin's existing target}" />
        </rule>
      </rules>
    </rewrite>

    <staticContent>
      <mimeMap fileExtension=".json" mimeType="application/json" />
    </staticContent>

  </system.webServer>
</configuration>
ภาษาไทย: ห้ามลบ rewrite rule เดิมที่แอดมินตั้งไว้ ให้เพิ่มต่อเท่านั้น ถ้า mimeMap ของ .json มีอยู่แล้วให้ข้าม ไม่งั้น IIS จะ error ว่าซ้ำ
7
Test it
ทดสอบ
  • Open index.html through http://as400webappdev/... (not by double-clicking the file)
  • Open DevTools → Console, confirm you see [IWS_CONFIG] ...
  • Call one endpoint and confirm you get JSON back, not an error
ภาษาไทย: ต้องเปิดผ่าน URL ของเว็บจริงเท่านั้น เปิดไฟล์ตรงๆ (double-click) จะเรียก API แบบ relative path ไม่ติด
8
Before going live — production checklist
ก่อนขึ้นระบบจริง — เช็กลิสต์
  • config.js → BASE_URL: point to the production path/server, not dev
  • config.js → AUTH: if login is required, don't hardcode passwords in JS — ask admin for a server-side proxy instead
  • web.config → CORS header (if set): change Access-Control-Allow-Origin "*" to your real domain
  • index.html → ?v= numbers: bump one more time before final release so users don't load stale cached files
  • Remove test/sample data such as placeholder empid or dummy records used during development
ภาษาไทย: จุดที่ต้องแก้จริงๆ ตอนขึ้นระบบมีแค่ไม่กี่จุด: URL ของ API, การเข้าสู่ระบบ (ถ้ามี), CORS, และเลข version ของไฟล์ — ที่เหลือไม่ต้องแตะถ้าทดสอบผ่านแล้ว
