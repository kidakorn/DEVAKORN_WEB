// Connection String สำหรับ PUB400.COM string connStr = "DataSource=pub400.com;UserID=X0250661;Password=YOUR_PWD;" + "DefaultCollection=X0250661;Naming=system;" + "ConnectionTimeout=30;MaxPoolSize=10;"; // ตัวเลือก Naming: // sql → ใช้ dot notation (X0250661.ITEM_MASTER) // system → ใช้ slash notation (X0250661/ITEM_MASTER)
💻 Code ตัวอย่างท้าย Week 3 — ต้องเขียนได้
// Week 3 — C# เชื่อม PUB400 ด้วย IBM.Data.Db2.iSeries using IBM.Data.Db2.iSeries; public class Db2Service { private readonly string _conn; public Db2Service(string connStr) => _conn = connStr; // ── Query: ดึง ITEM_MASTER ────────────────────── public async Task<List<ItemMaster>> GetItemsAsync() { var list = new List<ItemMaster>(); await using var cn = new iDB2Connection(_conn); await cn.OpenAsync(); await using var cmd = new iDB2Command( "SELECT CDITEM, NMITEM, QTONHAND, QTMIN FROM X0250661.ITEM_MASTER", cn); await using var rdr = await cmd.ExecuteReaderAsync(); while (await rdr.ReadAsync()) list.Add(new ItemMaster { CdItem = rdr.GetString(0).Trim(), NmItem = rdr.GetString(1).Trim(), QtOnhand = rdr.GetDecimal(2), QtMin = rdr.GetDecimal(3), }); return list; } // ── Call Stored Procedure: ISSUE_REQ ─────────── public async Task<(string result, string msg)> IssueReqAsync(string reqNo) { await using var cn = new iDB2Connection(_conn); await cn.OpenAsync(); await using var cmd = new iDB2Command("CALL X0250661.ISSUE_REQ(?, ?, ?)", cn); cmd.CommandType = CommandType.StoredProcedure; cmd.Parameters.Add("P_REQNO", iDB2DbType.iDB2Char, 10).Value = reqNo; var pResult = cmd.Parameters.Add("P_RESULT", iDB2DbType.iDB2Char, 1); var pMsg = cmd.Parameters.Add("P_ERRMSG", iDB2DbType.iDB2VarChar, 100); pResult.Direction = pMsg.Direction = ParameterDirection.Output; await cmd.ExecuteNonQueryAsync(); return (pResult.Value.ToString()!, pMsg.Value.ToString()!); } }
