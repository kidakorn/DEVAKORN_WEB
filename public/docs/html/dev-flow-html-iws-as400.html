Step 1.1 — สำรวจ IWS Endpoint ที่มี

ดูว่า RPG/SP ที่ Admin deploy ไว้มีอะไรบ้าง
IWS Endpoints ที่มีอยู่แล้ว (จาก RPG/SP)
# GET — ดึงข้อมูล GET /web/services/getStock → ITEM_MASTER list GET /web/services/getReqList → REQ_HEADER list GET /web/services/getReqDetail → REQ_DETAIL by REQ_NO GET /web/services/getPoList → PO_HEADER list GET /web/services/getVendorList → VENDOR_MASTER list # POST — business logic POST /web/services/issueReq → ISSUE_REQ SP POST /web/services/approveReq → APPROVE_REQ SP POST /web/services/returnReq → RETURN_REQ SP POST /web/services/receivePo → RECEIVE_PO SP

💡 ทดสอบ Endpoint ก่อนเขียน UI

  • เปิด Browser พิมพ์ URL GET endpoint โดยตรง — ต้องเห็น JSON response
  • ใช้ Postman หรือ Thunder Client (VS Code) ทดสอบ POST endpoint
  • ถ้าเห็น JSON ได้ = IWS ทำงานปกติ พร้อมเขียน Frontend

api.js — ฟังก์ชัน call IWS ทั้งหมด

ทุกหน้าเรียกผ่านไฟล์นี้ ไม่ fetch ตรงในหน้า
// js/api.js // ── Core fetch wrapper ────────────────────────── async function iwsGet(service, params = {}) { const qs = new URLSearchParams(params).toString(); const url = `${CONFIG.API_BASE}/${service}${qs ? '?'+qs : ''}`; const res = await fetch(url); if (!res.ok) throw new Error(`HTTP ${res.status}`); return res.json(); } async function iwsPost(service, body = {}) { const res = await fetch(`${CONFIG.API_BASE}/${service}`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(body) }); if (!res.ok) throw new Error(`HTTP ${res.status}`); return res.json(); } // ── Stock ──────────────────────────────────────── const API = { stock: { getAll: () => iwsGet('getStock'), getById: (code) => iwsGet('getStock', { CDITEM: code }), }, // ── Requisition ────────────────────────────────── req: { getList: (status) => iwsGet('getReqList', { STATUS: status || '' }), getDetail: (reqNo) => iwsGet('getReqDetail', { P_REQNO: reqNo }), issue: (reqNo) => iwsPost('issueReq', { P_REQNO: reqNo }), approve: (reqNo, by)=> iwsPost('approveReq', { P_REQNO: reqNo, P_APPROVED_BY: by }), return_: (reqNo) => iwsPost('returnReq', { P_REQNO: reqNo }), }, // ── Purchase Order ─────────────────────────────── po: { getList: (status) => iwsGet('getPoList', { STATUS: status || '' }), receive: (poNo, itemCode, qty) => iwsPost('receivePo', { P_PO_NO: poNo, P_ITEM_CODE: itemCode, P_QTY: qty }), }, // ── Vendor ─────────────────────────────────────── vendor: { getAll: () => iwsGet('getVendorList'), } };

// js/stock.js let allItems = []; // ── โหลดข้อมูลตอนเปิดหน้า ────────────────────── document.addEventListener('DOMContentLoaded', loadStock); async function loadStock() { show('loading'); hide('stock-table'); hide('error'); try { allItems = await API.stock.getAll(); renderTable(allItems); show('stock-table'); hide('loading'); } catch (err) { showError('โหลดข้อมูลไม่ได้: ' + err.message); } } // ── Render ────────────────────────────────────── function renderTable(items) { document.getElementById('stock-body').innerHTML = items.map(i => ` <tr> <td>${i.CDITEM}</td> <td>${i.NMITEM}</td> <td>${i.TXUNIT}</td> <td>${i.QTONHAND.toLocaleString()}</td> <td>${i.QTMIN.toLocaleString()}</td> <td><span class="badge ${i.QTONHAND <= i.QTMIN ? 'badge-low' : 'badge-approved'}"> ${i.QTONHAND <= i.QTMIN ? '⚠ ต่ำ' : 'ปกติ'} </span></td> </tr>`).join(''); } // ── Search filter ─────────────────────────────── function filterTable() { const q = document.getElementById('search').value.toLowerCase(); renderTable(allItems.filter(i => i.CDITEM.toLowerCase().includes(q) || i.NMITEM.toLowerCase().includes(q) )); } // ── Helpers ───────────────────────────────────── const show = id => document.getElementById(id).style.display = ''; const hide = id => document.getElementById(id).style.display = 'none'; function showError(msg) { const el = document.getElementById('error'); el.textContent = msg; el.style.display = ''; hide('loading'); }

⚠ ปัญหาที่เจอบ่อยตอนทดสอบ

  • CORS Error — ตอน localhost fetch ไป IWS บน IBM i จะโดน block ให้ใช้ Postman ทดสอบ API ก่อน แล้วให้ Admin เพิ่ม CORS header ใน httpd.conf
  • API ยังไม่พร้อม — Mock data ใน JS ไว้ก่อนเพื่อพัฒนา UI ต่อได้
  • Field name ผิด — ตรวจสอบชื่อ field จาก JSON response ของ IWS ให้ตรงกับที่ใช้ใน JS
