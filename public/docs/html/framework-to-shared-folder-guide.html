เตรียมเครื่อง (ทำครั้งเดียว)
  • ติดตั้ง Node.js (เวอร์ชัน LTS) บนเครื่องตัวเอง
  • เช็คว่าติดตั้งสำเร็จ: เปิด terminal พิมพ์ node -v ต้องเห็นเลขเวอร์ชัน
  • ขั้นตอนทั้งหมดด้านล่างทำบนเครื่องตัวเอง ไม่ใช่ในโฟลเดอร์แชร์ (โฟลเดอร์แชร์รับได้แค่ผลลัพธ์สุดท้าย)
ภาษาไทย: โฟลเดอร์แชร์ไม่มี Node.js รันอยู่ ต้อง build เสร็จก่อนบนเครื่องตัวเองแล้วค่อยเอาผลลัพธ์ไปวาง

2
เชื่อม IWS (โครงสร้างเดียวกับที่เคยทำ)
src/
  ├── config.js
  ├── api.js
  └── App.jsx 
// src/config.js — เหมือนเดิมทุกอย่าง
export const IWS_CONFIG = {
  BASE_URL: "/api/web/serveices",
  ENDPOINTS: {
    GET_DATA: "/scanner/minpop"
  }
};
// src/api.js
import { IWS_CONFIG } from './config';

async function handleResponse(res) {
  const raw = await res.text();
  if (!res.ok) throw new Error(`HTTP ${res.status}: ${raw}`);
  return JSON.parse(raw);
}

export async function getData(empid) {
  const url = IWS_CONFIG.BASE_URL + IWS_CONFIG.ENDPOINTS.GET_DATA + "?empid=" + empid;
  const res = await fetch(url, { headers: { "Accept": "application/json" } });
  return handleResponse(res);
}
// src/App.jsx
import { useState } from 'react';
import { getData } from './api';

function App() {
  const [empid, setEmpid] = useState("25066");
  const [rows, setRows] = useState([]);

  async function loadData() {
    const data = await getData(empid);
    setRows(Array.isArray(data) ? data : [data]);
  }

  return (
    <div>
      <input value={empid} onChange={e => setEmpid(e.target.value)} />
      <button onClick={loadData}>Load Data</button>
      <pre>{JSON.stringify(rows, null, 2)}</pre>
    </div>
  );
}

export default App;
ภาษาไทย: concept เดียวกับ config.js/api.js ที่ทำกันมาตลอด แค่เปลี่ยนจาก <script> ธรรมดาเป็น import/export แบบ module เท่านั้น

2
ตั้งค่า Static Export
// next.config.js
/** @type {import('next').NextConfig} */
const nextConfig = {
  output: 'export',
  images: { unoptimized: true }   // จำเป็นเพราะ static export ใช้ Image Optimization ไม่ได้
};

module.exports = nextConfig;
🔴 ห้ามใช้: API Routes (/app/api/...), Server Actions, Middleware — ทั้งหมดนี้ต้องมี server รันตลอดเวลา ซึ่งโฟลเดอร์แชร์ไม่มี
3
เชื่อม IWS (ต้องเป็น Client Component)
// app/config.js
export const IWS_CONFIG = {
  BASE_URL: "/api/web/serveices",
  ENDPOINTS: { GET_DATA: "/scanner/minpop" }
};
// app/page.jsx
'use client';   // 🔴 ต้องมีบรรทัดนี้ ไม่งั้น fetch ฝั่ง client จะไม่ทำงานตอน export

import { useState } from 'react';
import { IWS_CONFIG } from './config';

export default function Home() {
  const [empid, setEmpid] = useState("25066");
  const [rows, setRows] = useState([]);

  async function loadData() {
    const url = IWS_CONFIG.BASE_URL + IWS_CONFIG.ENDPOINTS.GET_DATA + "?empid=" + empid;
    const res = await fetch(url, { headers: { "Accept": "application/json" } });
    const data = await res.json();
    setRows(Array.isArray(data) ? data : [data]);
  }

  return (
    <div>
      <input value={empid} onChange={e => setEmpid(e.target.value)} />
      <button onClick={loadData}>Load Data</button>
      <pre>{JSON.stringify(rows, null, 2)}</pre>
    </div>
  );
}
ภาษาไทย: 'use client' สำคัญมาก — ถ้าลืม Next.js จะพยายาม fetch ตอน build (ฝั่ง server) ซึ่งจะพังเพราะตอน build ไม่มี IWS ให้เรียก
4
Build + วางในโฟลเดอร์แชร์
npx next build

จะได้โฟลเดอร์ out/ — copy ไฟล์ข้างในไปวางใน webas400app/kidakorn/ เหมือนทาง React

web.config — ต่างกันตามแบบที่เลือก

รูปแบบต้องทำอะไร
Next.js Static Exportไม่ต้องเพิ่ม SPA fallback — แต่ละหน้ามีไฟล์ .html จริงแยกกันอยู่แล้ว
React + React Router (SPA)ต้องเพิ่ม <conditions> ให้ทุก route fallback ไป index.html (ตามที่คุยกันไว้ก่อนหน้า)
React แบบหน้าเดียว ไม่มี routerไม่ต้องทำอะไรเพิ่ม

Troubleshooting

อาการสาเหตุวิธีแก้
หน้าเปิดขึ้นแต่ CSS/JS โหลดไม่ขึ้น (หน้าขาวๆ)path ของไฟล์ผิด เพราะ build มาแบบ absolute pathReact: ตั้ง base: './' ใน vite.config.js
Next.js build error เกี่ยวกับ Imageยังไม่ได้ปิด Image Optimizationเพิ่ม images: { unoptimized: true }
fetch ไม่ทำงานเลยตอนเปิดหน้าเว็บจริงลืมใส่ 'use client' (Next.js เท่านั้น)เพิ่มบรรทัดนี้บนสุดของไฟล์ component
404 ตอนเปลี่ยนหน้า (React Router)web.config ยังไม่มี SPA fallbackเพิ่ม <conditions> ตามตารางด้านบน
